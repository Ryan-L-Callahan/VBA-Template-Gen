Sub hyperlinks()

    Dim rng As Range, cell As Range, x As Integer
    Set rng = Range("'Layout'!D3:D1000")
    For Each cell In rng
        If cell.Value <> "" Then
            cell.Select
            Worksheets("Layout").hyperlinks.Add Anchor:=Selection, Address:="", SubAddress:= _
            cell.Value
        End If
    Next cell
End Sub
